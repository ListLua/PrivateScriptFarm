local UserInputService = game:GetService("UserInputService")
local GuiService = game:GetService("GuiService")
local player = game.Players.LocalPlayer

local Drag = false

local DragStartPosition

local UI = script.Parent
local SelectionBox = UI.SelectionBox

local function FindUIsInSelectionBox()
	local selPos = SelectionBox.AbsolutePosition
	local selSize = SelectionBox.AbsoluteSize

	local selMin = selPos
	local selMax = selPos + selSize

	for _, v in UI:GetDescendants() do
		if v:IsA("Frame") and v.Visible and v ~= SelectionBox then
			local uiPos = v.AbsolutePosition
			local uiSize = v.AbsoluteSize

			local uiMin = uiPos
			local uiMax = uiPos + uiSize

			local overlapX = math.min(selMax.X, uiMax.X) - math.max(selMin.X, uiMin.X)
			local overlapY = math.min(selMax.Y, uiMax.Y) - math.max(selMin.Y, uiMin.Y)

			if overlapX > 0 and overlapY > 0 then
				v.BackgroundColor3 = Color3.new(0,1,0)
			else
				v.BackgroundColor3 = Color3.new(1,1,1)
			end
		end
	end
end


UserInputService.InputBegan:Connect(function(InputObject, gameProcessedEvent : boolean)
	if gameProcessedEvent then return end
	
	if InputObject.UserInputType == Enum.UserInputType.MouseButton1 then
		local MouseDelta = UserInputService:GetMouseLocation()
		
		Drag = true
		DragStartPosition = UDim2.fromOffset(MouseDelta.X, MouseDelta.Y - GuiService:GetGuiInset().Y)
		
		SelectionBox.BackgroundTransparency = 0.5
		SelectionBox.Size = UDim2.fromScale(0, 0)
		SelectionBox.Position = DragStartPosition
	end
end)

UserInputService.InputEnded:Connect(function(InputObject, gameProcessedEvent : boolean)
	if gameProcessedEvent then return end
	
	if InputObject.UserInputType == Enum.UserInputType.MouseButton1 then
		Drag = false
		SelectionBox.BackgroundTransparency = 1
		SelectionBox.Position = UDim2.fromScale(0, 0)
		FindUIsInSelectionBox()
	end
end)

UserInputService.InputChanged:Connect(function(InputObject, gameProcessedEvent)
--	if gameProcessedEvent then return end
	
	if InputObject.UserInputType == Enum.UserInputType.MouseMovement then
		if Drag and DragStartPosition then
			local MouseDelta = UserInputService:GetMouseLocation()
			
			local PositionChange = UDim2.fromOffset(MouseDelta.X, MouseDelta.Y - GuiService:GetGuiInset().Y) - DragStartPosition 
			
			SelectionBox.Size = UDim2.fromOffset(math.abs(PositionChange.X.Offset), math.abs(PositionChange.Y.Offset))
			SelectionBox.Position = UDim2.fromOffset(math.min(DragStartPosition.X.Offset, MouseDelta.X), math.min(DragStartPosition.Y.Offset, MouseDelta.Y))
			FindUIsInSelectionBox()
		end
	end
end)
